class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        left = 0
        right = len(nums) - 1
        ans = 0
        
        while left <= right:
            mid = (left + right) // 2
            
            
            if nums[mid] < target:
                left = mid + 1
                ans = left
            elif nums[mid] > target:
                right = mid - 1
            else:
                return mid
            
        return ans
