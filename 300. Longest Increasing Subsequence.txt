class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        l = len(nums)
        
        dp = [1 for _ in range(l)]
        
        for i in range(l):
            for j in range(i, l):
                if nums[i] < nums[j]:
                    dp[j] = max(dp[j], dp[i] + 1)
                    
#        print(dp)                   
        return max(dp)
